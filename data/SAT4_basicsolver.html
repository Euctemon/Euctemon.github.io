Conjunctive normal form

When we chose the alphabet for propositional logic, we used the set \( (\bot, \land, \lor, \to) \). But modern solvers use different set as the alphabet, namely \( (\bot, \top, \land, \lor) \). There are two questions connected to the chenge of alphabet. Why would we want a different alphabet a do these two alphabets generate the same set of formulas? To show we can generate the same set of formulas using these alphabets, we need to show that every formula generated by the first alphabet can be generated by the second alphabet and vice versa. We know that symbols \(\bot,\land,\lor\) are in both alphabets. This means we need to show we can write implication using the second alphabet and negation using the first alphabet. Negation \(\lnot A\) is defined as \(a \to \bot\). This means we only need to show how to write implication using the second alphabet. \(\dfrac{\dfrac{}{P \to Q \vdash P \lor \lnot P} \vdash \qquad \dfrac{\dfrac{\dfrac{}{P \to Q \vdash P \to Q} \qquad \dfrac{}{P \vdash P}}{P \to Q, P \vdash Q} \to_E}{P \to Q, P \vdash \lnot P \lor Q} \lor_I \qquad \dfrac{\dfrac{\dfrac{}{\lnot P \vdash \lnot P}}{P \to Q, \lnot P \vdash \lnot P} w_I}{P \to Q, \lnot P \vdash \lnot P \lor Q} \lor_I}{P \to Q \vdash \lnot P \lor Q} \lor_E\) # \dfrac{\dfrac{}{\lnot P \lor Q, P \vdash \lnot P \lor Q} \vdash \qquad \dfrac{}{\lnot P \lor Q, P, \lnot P \vdash Q} \vdash \qquad \dfrac{\dfrac{}{Q \vdash Q}}{\lnot P \lor Q, P, Q \vdash Q} w_I}{\lnot \lor Q, P \vdash Q}

Let \(\mathfrac{A} = (\bot,\land)\)
For every set \(\Gamma\) of propositional formulas and every propositional formula \(A\), we have that \(\Gamma \vdash A\) implies \(\Gamma \vDash A\). 

type formula =
  Const of bool
| Var of int
| And of formula * formula
| Or of formula * formula
| Imp of formula * formula



type sat = Sat | Unsat



let rec string_of_formula f = match f with
| Const b -> string_of_bool b
| Var i -> string_of_int i
| And (f_left,f_right) -> String.concat "" ["("; string_of_formula f_left; " and "; string_of_formula f_right; ")"]
| Or (f_left,f_right) -> String.concat "" ["("; string_of_formula f_left; " or "; string_of_formula f_right; ")"]
| Imp (f_left,f_right) -> String.concat "" ["("; string_of_formula f_left; " implies "; string_of_formula f_right; ")"]



let string_of_sat s = match s with
| Sat -> "satisfiable"
| Unsat -> "unsatisfiable"



let rec free_var f = match f with
| Const _ -> None
| Var i -> Some i
| And (f_left, f_right) | Or (f_left, f_right) | Imp (f_left, f_right) ->
	let left_var = free_var f_left in
	if Option.is_some left_var then left_var else free_var f_right



let rec assign_var f i b = match f with
| Var j -> if i = j then Const b else Var j
| And (f_left, f_right) -> And (assign_var f_left i b, assign_var f_right i b)
| Or (f_left, f_right) -> Or (assign_var f_left i b, assign_var f_right i b)
| Imp (f_left, f_right) -> Imp (assign_var f_left i b, assign_var f_right i b)
| other -> other



let simp_and f_left f_right = match (f_left, f_right) with
| (Const true, _) -> f_right
| (_, Const true) -> f_left
| (Const false, _) | (_, Const false) -> Const false
| _ -> And (f_left, f_right)



let simp_or f_left f_right = match (f_left, f_right) with
| (Const false, _) -> f_right
| (_, Const false) -> f_left
| (Const true, _) | (_, Const true) -> Const true
| _ -> Or (f_left, f_right)



let simp_imp f_left f_right = match (f_left, f_right) with
| (Const true, _) -> f_right
| (Const false, _) | (_, Const true) -> Const true
| _ -> Imp (f_left, f_right)



let rec simp_formula f = match f with
| And (f_left, f_right) -> simp_and (simp_formula f_left) (simp_formula f_right)
| Or (f_left, f_right) -> simp_or (simp_formula f_left) (simp_formula f_right)
| Imp (f_left, f_right) -> simp_imp (simp_formula f_left) (simp_formula f_right)
| other -> other



let rec satisfiable f = match free_var f with
| None -> if (simp_formula f = Const true) then Sat else Unsat
| Some i ->
  let true_guess = simp_formula (assign_var f i true) in
  let false_guess = simp_formula (assign_var f i false) in
    if satisfiable true_guess = Sat || satisfiable false_guess = Sat then Sat else Unsat
